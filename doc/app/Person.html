def self.marriage_strategy (m,f) 
  (m.age / 2 + 7) < f.age and 
    (f.age / 2 + 7) < m.age and 
    not m.married? and 
    not f.married? and 
    not f.gender == m.gender and
    # not f.sibling_of? m and
    # not f.aunt_or_uncle_of? m and
    # not f.niece_or_nephew_of? m and
    # not f.cousin_of? m
    f.age > @@coming_of_age and
    m.age > @@coming_of_age
end
